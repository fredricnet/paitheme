{{/* Tag Badge Shortcode - Creates a badge/tag with icons and text Parameters: - text: Text to display in the tag (required) - will be capitalized (first letter only) - icon: Icon type (optional, default: "star") - icon-count: Number of icons to display (optional, default: 5) - icon-color: Color of icons (optional, default: "#3b82f6") Usage example: tag-badge text="real results" icon="star" icon-count="5" icon-color="#3b82f6" */}} {{- $text := .Get "text" -}} {{- $iconType := .Get "icon" | default "star" -}} {{- $iconCount := .Get "icon-count" | default 5 -}} {{- $iconColor := .Get "icon-color" | default "#3b82f6" -}} {{- if $text -}} {{- $firstChar := substr $text 0 1 | upper -}} {{- $rest := substr $text 1 -}} {{- $capitalizedText := printf "%s%s" $firstChar $rest -}}
{{- range seq $iconCount -}} {{- if eq $iconType "star" -}} {{- end -}} {{- end -}} {{ $capitalizedText }}
{{- end -}}
